import java.util.*;
class Node{
    int data;
    Node next;
    Node(int data){
        this.data=data;
    }
} 
class List{
    Node head,tail;
    public void insert(int data){//Insertion
        Node n=new Node(data);
        if(head==null){
            head=n;
            tail=n;
        }
        else{
            tail.next=n;
            tail=n;
        }
    }
    public void display(){//Display
        if(head==null){
            System.out.println("Empty");
            return;
        }
        Node temp=head;
        System.out.println("Card Transactions:");
        while(temp!=null){
            System.out.println(temp.data);
            temp=temp.next;
        }
    }
    public void points(){//Sum of bill amount 
        int s=0;
        Node temp=head;
        while(temp!=null){
            s+=temp.data;
            temp=temp.next;
        }
        System.out.println("Gained Points:"+s);
    }
}
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		List l=new List();
		int i=1;
		while(i>0 && i<4){
		    System.out.println("1.Visit the food court");
            System.out.println("2.View my transactions");
            System.out.println("3.Show my points");
            System.out.println("Enter your choice");
            int ch=sc.nextInt();
            switch(ch){
                case 1:
                    System.out.println("Pay bill amount:");
                    int data=sc.nextInt();
                    l.insert(data);
                    break;
                case 2:
                    l.display();
                    break;
                case 3:
                    l.points();
                    break;
                default:
                    System.out.println("Visit again");
                    return;
            }
		}
	}
}
